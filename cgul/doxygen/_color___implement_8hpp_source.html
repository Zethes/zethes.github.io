12 static CGUL::Float32 HueToRGB(CGUL::Float32 p, CGUL::Float32 q, CGUL::Float32 t)
13 {
14  if (t < 0.0f)
15  {
16  t += 1;
17  }
18  if (t > 1.0f)
19  {
20  t -= 1;
21  }
22  if (t < 1.0f / 6.0f)
23  {
24  return p + (q - p) * 6.0f * t;
25  }
26  if (t < 1.0f / 2.0f)
27  {
28  return q;
29  }
30  if (t < 2.0f / 3.0f)
31  {
32  return p + (q - p) * (2.0f / 3.0f - t) * 6.0f;
33  }
34  return p;
35 }

49  {
50  r = g = b = luminance;
51  }
52  else
53  {
54  hue /= 360.0f;
55  Float32 q = luminance < 0.5 ? luminance * (1.0f + saturation) : luminance + saturation - luminance * saturation;

59  b = HueToRGB(p, q, hue - 1.0f / 3.0f) ;
60  }

294 _CGUL_INLINE_DEFINE void CGUL::ColorT< Type >::GetHSL(Float32* hue, Float32* saturation, Float32* luminance) const
295 {
296  Float32 red = r / (Float32)maxValue;
297  Float32 green = g / (Float32)maxValue;
298  Float32 blue = b / (Float32)maxValue;
299  Float32 max = Math::Max(red, green, blue);
300  Float32 min = Math::Min(red, green, blue);
301  *luminance = (max + min) / 2.0f;
302 
303  if(max == min)
304  {
305  *hue = 0.0f;
306  *saturation = 0.0f;
307  }
308  else
309  {
310  Float32 d = max - min;
311  *saturation = *luminance > 0.5f ? d / (2.0f - max - min) : d / (max + min);
312  if (max == red)
313  {
314  *hue = (green - blue) / d + (green < blue ? 6.0f : 0.0f);
315  }
316  else if (max == green)
317  {
318  *hue = (blue - red) / d + 2.0f;
319  }
320  else
321  {
322  *hue = (red - green) / d + 4.0f;
323  }
324  *hue = *hue / 6.0f;
325  }
