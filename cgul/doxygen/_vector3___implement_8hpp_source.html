575 template< typename Type >
576 _CGUL_INLINE_IMPLEMENT void CGUL::Vector3T< Type >::MakeOrthonormalBasis(Vector3T< Type >* vectorB, Vector3T< Type >* vectorC)
577 {
578  Vector3T< Type > vectorA(*this);
579  if (vectorA.GetManhattanMagnitude() == 0)
580  {
581  return;
582  }
583  vectorA.Normalize();
584  vectorB->Set(1, 0, 0);
585  *vectorC = CrossProduct(vectorA, *vectorB);
586  if (vectorC->GetManhattanMagnitude() == 0)
587  {
588  vectorB->Set(0, 1, 0);
589  *vectorC = CrossProduct(vectorA, *vectorB);
590  }
591  *vectorB = CrossProduct(vectorA, *vectorC);
592  vectorB->Normalize();
593  vectorC->Normalize();
594 }
