635 template< typename Type >
636 _CGUL_INLINE_IMPLEMENT void CGUL::Vector3T< Type >::MakeOrthonormalBasis(Vector3T< Type >* vectorB, Vector3T< Type >* vectorC)
637 {
638  Vector3T< Type > vectorA(*this);
639  if (vectorA.GetManhattanMagnitude() == 0)
640  {
641  return;
642  }
643  vectorA.Normalize();
644  vectorB->Set(1, 0, 0);
645  *vectorC = CrossProduct(vectorA, *vectorB);
646  if (vectorC->GetManhattanMagnitude() == 0)
647  {
648  vectorB->Set(0, 1, 0);
649  *vectorC = CrossProduct(vectorA, *vectorB);
650  }
651  *vectorB = CrossProduct(vectorA, *vectorC);
652  vectorB->Normalize();
653  vectorC->Normalize();
654 }
