356 template< typename Type >
357 _CGUL_INLINE_IMPLEMENT Type CGUL::Math::StandardDeviation(const FixedList< Type >& values, Type* mean)
358 {
359  Type average = 0;
360  for (Size i = 0, iEnd = values.GetSize(); i < iEnd; ++i)
361  {
362  average += values.Get(i);
363  }
364  average /= values.GetSize();
365 
366  if (mean != NULL)
367  {
368  *mean = average;
369  }
370 
371  Type variance = 0;
372  for (Size i = 0, iEnd = values.GetSize(); i < iEnd; ++i)
373  {
374  variance += Math::Sqr(values.Get(i) - average);
375  }
376  variance /= values.GetSize();
377  return Math::Sqrt(variance);
378 }
